00374 
00375   //Check if done
00376   if (!mIsInitialised) {

00380   //Shutdown teleop source adapter (will stop if necessary)

00386   if (NULL != mTeleopSource) {
00387     delete mTeleopSource;
00388     mTeleopSource = NULL;
00389   } else {
00390     ROS_WARN("TeleopSourceNode::shutdown: teleop source already deleted");

00392 
00393   //Zero message
00394   for (size_t i = 0; i < mTeleopStateMsg.axes.size(); i++) {
00395     mTeleopStateMsg.axes[i].value = 0.0;
00396   }
00397   for (size_t i = 0; i < mTeleopStateMsg.buttons.size(); i++) {
00398     mTeleopStateMsg.buttons[i].value = 0;
00399   }
00400 
00401   //Publish zero message
00402   mPublisher.publish(mTeleopStateMsg);

00404   //Sleep a bit to allow zero message to be published before node is shutdown.
00405   //Use boost thread sleep rather than ROS sleep since ROS sleep may try to use

00419 teleop::TeleopSource* TeleopSourceNode::teleopSourceFactory() {
00420   teleop::TeleopSource* teleopSource = NULL;
00421   std::string teleopType = mTeleopType;
00422 
00423   //If type is auto, check for best alternative
00424   if (0 == teleopType.compare(std::string(TELEOP_TYPE_AUTO))) {
00425     teleop::TeleopSourceJoystick* teleopSourceJoystick;
00426     teleopSourceJoystick = new teleop::TeleopSourceJoystick();
00427     teleopSourceJoystick->setDevice(mJoystickDevice);
00428     if (teleopSourceJoystick->init()) {
00429       if (!teleopSourceJoystick->shutdown()) {
00430         ROS_WARN("TeleopSourceNode::teleopSourceFactory: ignoring error during TeleopSourceJoystick shutdown");
00431       }
00432       teleopType = std::string(TELEOP_TYPE_JOYSTICK);
00433     } else {
00434       teleopType = std::string(TELEOP_TYPE_KEYBOARD);
00435     }
00436     delete teleopSourceJoystick;
00437   }
00438 
00439   //Handle known types
00440   if (0 == teleopType.compare(std::string(TELEOP_TYPE_KEYBOARD))) {
00441     teleop::TeleopSourceKeyboard* teleopSourceKeyboard;
00442     teleopSourceKeyboard = new teleop::TeleopSourceKeyboard();
00443     if (!teleopSourceKeyboard->setSteps((unsigned int)mKeyboardSteps)) {
00444       ROS_WARN("TeleopSourceNode::teleopSourceFactory: unable to set keyboard steps");
00445     }
00446     teleopSource = teleopSourceKeyboard;
00447   } else if (0 == teleopType.compare(std::string(TELEOP_TYPE_JOYSTICK))) {
00448     teleop::TeleopSourceJoystick* teleopSourceJoystick;
00449     teleopSourceJoystick = new teleop::TeleopSourceJoystick();
00450     if (!teleopSourceJoystick->setDevice(mJoystickDevice)) {
00451       ROS_WARN("TeleopSourceNode::teleopSourceFactory: unable to set joystick device");
00452     }
00453     teleopSource = teleopSourceJoystick;
00454   } else {
00455     ROS_ERROR("TeleopSourceNode::teleopSourceFactory: unknown teleop source type (%s)", teleopType.c_str());
00456     return NULL;
00457   }
00458 
00459   //Return result
00460   return teleopSource;

00475     mTeleopStateMsg.buttons.resize(teleopState->buttons.size());

00477   for (size_t i = 0; i < mTeleopStateMsg.axes.size(); i++) {
00478     mTeleopStateMsg.axes[i].type = teleopState->axes[i].type;
00479     mTeleopStateMsg.axes[i].value = teleopState->axes[i].value;
00480   }
00481   for (size_t i = 0; i < mTeleopStateMsg.buttons.size(); i++) {
00482     mTeleopStateMsg.buttons[i].type = teleopState->buttons[i].type;
00483     mTeleopStateMsg.buttons[i].value = teleopState->buttons[i].value;

00485

00488
