00644 bool TeleopSinkTwistNode::teleopStateToTwist(const teleop_msgs::State* const teleopStateMsg,
00645                                              geometry_msgs::Twist* const twistMsg) {
00646   //Sanity check parameters
00647   if (NULL == teleopStateMsg || NULL == twistMsg) {
00648     ROS_ERROR("teleopStateToTwist: NULL parameter");
00649     return false;
00650   }
00651 
00652   //Get indices for each relevant axis in teleop state
00653   int indexLinX = -1;
00654   int indexLinY = -1;
00655   int indexLinZ = -1;
00656   int indexRotX = -1;
00657   int indexRotY = -1;
00658   int indexRotZ = -1;
00659   int indexThrottle = -1;
00660   for (size_t i = 0; i < teleopStateMsg->axes.size(); i++) {
00661     switch (teleopStateMsg->axes[i].type) {
00662       case teleop::TELEOP_AXIS_TYPE_LIN_X:    indexLinX = i;     break;
00663       case teleop::TELEOP_AXIS_TYPE_LIN_Y:    indexLinY = i;     break;
00664       case teleop::TELEOP_AXIS_TYPE_LIN_Z:    indexLinZ = i;     break;
00665       case teleop::TELEOP_AXIS_TYPE_ROT_X:    indexRotX = i;     break;
00666       case teleop::TELEOP_AXIS_TYPE_ROT_Y:    indexRotY = i;     break;
00667       case teleop::TELEOP_AXIS_TYPE_ROT_Z:    indexRotZ = i;     break;
00668       case teleop::TELEOP_AXIS_TYPE_THROTTLE: indexThrottle = i; break;
00669       default:
00670         //Ignore other axes
00671         break;
00672     }
00673   }
00674 
00675   //Find throttle value
00676   double throttle = 1.0;
00677   if (-1 != indexThrottle) {
00678     throttle = teleopStateMsg->axes[indexThrottle].value;
00679   }
00680 
00681   //Set applicable twist message values for each axis using teleop axis values.
00682   //Teleop axes can be mapped to various twist axes, depending on which axes
00683   //the sink object has and which axes the teleop source provides.  Each teleop
00684   //axis is mapped according to the following priority scheme.
00685   //
00686   // ---------------
00687   // teleop -> twist
00688   // ---------------
00689   // LIN_X  -> linear.x (otherwise angular.y or angular.z)
00690   // LIN_Y  -> linear.y (otherwise angular.x or angular.z)
00691   // LIN_Z  -> linear.z (otherwise angular.z)
00692   // ROT_X  -> angular.x (otherwise linear.y or linear.z)
00693   // ROT_Y  -> angular.y (otherwise linear.x or linear.z)
00694   // ROT_Z  -> angular.z (otherwise linear.z)
00695   // ---------------
00696 
00697   //Handle lin X
00698   if (-1 != indexLinX) {
00699     if (mHasLinX) {
00700       twistMsg->linear.x = teleopToTwistLinX(throttle, teleopStateMsg->axes[indexLinX].value);
00701     } else if (mHasRotY && -1 == indexRotY) {
00702       twistMsg->angular.y = teleopToTwistRotY(throttle, teleopStateMsg->axes[indexLinX].value);
00703     } else if (mHasRotZ && -1 == indexRotZ) {
00704       twistMsg->angular.z = teleopToTwistRotZ(throttle, teleopStateMsg->axes[indexLinX].value);
00705     }
00706   }
00707 
00708   //Handle lin Y
00709   if (-1 != indexLinY) {
00710     if (mHasLinY) {
00711       twistMsg->linear.y = teleopToTwistLinY(throttle, teleopStateMsg->axes[indexLinY].value);
00712     } else if (mHasRotX && -1 == indexRotX) {
00713       twistMsg->angular.x = teleopToTwistRotX(throttle, teleopStateMsg->axes[indexLinY].value);
00714     } else if (mHasRotZ && -1 == indexRotZ) {
00715       twistMsg->angular.z = teleopToTwistRotZ(throttle, teleopStateMsg->axes[indexLinY].value);
00716     }
00717   }
00718 
00719   //Handle lin Z
00720   if (-1 != indexLinZ) {
00721     if (mHasLinZ) {
00722       twistMsg->linear.z = teleopToTwistLinZ(throttle, teleopStateMsg->axes[indexLinZ].value);
00723     } else if (mHasRotZ && -1 == indexRotZ) {
00724       twistMsg->angular.z = teleopToTwistRotZ(throttle, teleopStateMsg->axes[indexLinZ].value);
00725     }
00726   }
00727 
00728   //Handle rot X
00729   if (-1 != indexRotX) {
00730     if (mHasRotX) {
00731       twistMsg->angular.x = teleopToTwistRotX(throttle, teleopStateMsg->axes[indexRotX].value);
00732     } else if (mHasLinY && -1 == indexLinY) {
00733       twistMsg->linear.y = teleopToTwistLinY(throttle, teleopStateMsg->axes[indexRotX].value);
00734     } else if (mHasLinZ && -1 == indexLinZ) {
00735       twistMsg->linear.z = teleopToTwistLinZ(throttle, teleopStateMsg->axes[indexRotX].value);
00736     }
00737   }
00738 
00739   //Handle rot Y
00740   if (-1 != indexRotY) {
00741     if (mHasRotY) {
00742       twistMsg->angular.y = teleopToTwistRotY(throttle, teleopStateMsg->axes[indexRotY].value);
00743     } else if (mHasLinX && -1 == indexLinX) {
00744       twistMsg->linear.x = teleopToTwistLinX(throttle, teleopStateMsg->axes[indexRotY].value);
00745     } else if (mHasLinZ && -1 == indexLinZ) {
00746       twistMsg->linear.z = teleopToTwistLinZ(throttle, teleopStateMsg->axes[indexRotY].value);
00747     }
00748   }
00749 
00750   //Handle rot Z
00751   if (-1 != indexRotZ) {
00752     if (mHasRotZ) {
00753       twistMsg->angular.z = teleopToTwistRotZ(throttle, teleopStateMsg->axes[indexRotZ].value);
00754     } else if (mHasLinZ && -1 == indexLinZ) {
00755       twistMsg->linear.z = teleopToTwistLinZ(throttle, teleopStateMsg->axes[indexRotZ].value);
00756     }
00757   }
00758 
00759   //Return result
00760   return true;
00761 }

00763 teleop::TeleopAxisValue TeleopSinkTwistNode::applyQuadratic(bool enabled, teleop::TeleopAxisValue teleopAxisValue) {
00764   if (enabled) {
00765     if (0.0 <= teleopAxisValue) {
00766       return (teleopAxisValue * teleopAxisValue);
00767     } else {
00768       return -(teleopAxisValue * teleopAxisValue);
00769     }
00770   } else {
00771     return teleopAxisValue;
00772   }
00773 }

00776                                                            double throttle,
00777                                                            teleop::TeleopAxisValue teleopAxisValue) {
00778   if (enabled) {
00779     return (throttle * teleopAxisValue);
00780   } else {
00781     return teleopAxisValue;
00782   }
00783 }
